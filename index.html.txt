  <script>
    let player = {
      name: "プレイヤー",
      exp: 0,
      equipment: "白衣",
      title: "見習い検査技師"
    };

    const savedPlayer = localStorage.getItem("player");
    if (savedPlayer) {
      player = JSON.parse(savedPlayer);
    } else {
      localStorage.setItem("player", JSON.stringify(player));
    }

    function getLevel(exp) {
      return Math.floor(exp / 100) + 1;
    }

    function getNextExp(exp) {
      return Math.floor(exp / 100) * 100 + 100;
    }

    function updateUI() {
      const level = getLevel(player.exp);
      const nextExp = getNextExp(player.exp);
      const currentExp = player.exp % 100;
      const percent = (currentExp / 100) * 100;

      document.getElementById("playerName").textContent = player.name;
      document.getElementById("playerLevel").textContent = level;
      document.getElementById("playerExp").textContent = player.exp;
      document.getElementById("nextExp").textContent = nextExp;
      document.getElementById("expBar").style.width = percent + "%";
    }

    function setMessage(text) {
      document.getElementById("messageBox").textContent = text;
    }

    function goQuest() {
      setMessage("クエスト画面へ進みます。あとで quest.html や Googleフォームへのリンクに変更できます。");
      // 例:
      // location.href = "quest.html";
      // または
      // window.open("GoogleフォームのURL", "_blank");
    }

    function goEquipment() {
      setMessage("装備画面へ進みます。称号、白衣、アイテム、分野別バッジなどを表示できます。");
      // 例:
      // location.href = "equipment.html";
    }

    function showStatus() {
      const level = getLevel(player.exp);
      setMessage(
        "名前：" + player.name +
        " / レベル：" + level +
        " / 総EXP：" + player.exp +
        " / 称号：" + (player.title || "なし")
      );
    }

    function showSaveData() {
      const exists = localStorage.getItem("player") ? "保存されています" : "未保存です";
      setMessage("セーブデータはブラウザ内に " + exists + "。スマホでも同じ端末なら保持されます。");
    }

    updateUI();
  </script>
